interface OverlayContextProps {
  open: (overlayElement: React.ReactNode) => void;
  close: () => void;
}

export const OverlayDispatchContext = createContext<OverlayContextProps | null>(null);

const OverlayContext = ({ children }: PropsWithChildren) => {
  const [overlayComponent, setOverlayComponent] = useState<React.ReactNode>(null);

  const open = (overlayElement: React.ReactNode) => {
    setOverlayComponent(overlayElement);
  };

  const close = () => {
    setOverlayComponent(null);
  };

  const dispatch = {
    open,
    close,
  };

  return (
    <OverlayDispatchContext.Provider value={dispatch}>
      {children}
      {overlayComponent}
    </OverlayDispatchContext.Provider>
  );
};

export default OverlayContext;
// src/App.tsx
export default function App() {
  return (
		<OverlayContext>
			// ...
		</OverlayContext>
  );
}
